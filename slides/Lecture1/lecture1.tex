		\end{quote}

	\begin{itemize}
		\item What this argument in fact shows is that the first-order formula 
		\begin{multline*}
			\forall x \forall y \big(x \neq y \rightarrow \big(r(x,y) \vee r(y,x)\big)\big) 
			\rightarrow\\
			\exists x \forall y \big(x \neq y \rightarrow 
			\big[r(x,y) \vee \exists z \big(r(x,z) \wedge  r(z,y)\big)\big]\big),
		\end{multline*}
		is true in {\color{blue} every finite structure}. 
		\item But this formula is not true in {\color{blue} every structure}: consider the natural numbers $\N$ for the domain of quantification, and let $r(x,y)$ be ``$x$ is greater than $y$''.  
	\end{itemize}
\end{frame}


\begin{frame}

			\item Does there exist an {\color{blue} algorithm} to determine whether a given formula of first-order logic is valid? 
			\item Does there exist an algorithm to determine whether it is finitely valid?

\begin{frame}
	\begin{itemize}	
		\item Usually, this problem is posed in a dual form.
		\item A formula $\phi(\bar{x})$ of first-order logic 
		is \cbl{satisfiable} if, for some structure $\fA$ and tuple of objects $\bar{a}$ from $A$, we have $\fA \models \phi[\bar{a}]$.
		\item A formula $\phi(\bar{x})$ is \cbl{finitely satisfiable} if, for some \cbl{finite} structure $\fA$ and tuple $\bar{a}$, we have $\fA \models \phi[\bar{a}]$.
		\item Thus, a sentence $\phi$ is (finitely) valid if and only if $\neg \phi$ is not (finitely) satisfiable.
		\item We apply the terms ``satisfiable'' and ``finitely satisfiable'' to \cbl{sets} of formulas $\Phi(\bar{x})$.
	\end{itemize}
\end{frame}


\begin{frame}
\begin{itemize}
\item We can think of any decision problem as a \cbl{set of strings}.
\item A set (of strings) $L$ is said to be \cbl{computable} or \cbl{recursive} if there is some Turing machine $M$ that is guaranteed to halt, and that terminates successfully on input $x$ if and only if $x \in L$.
\item A set $L$ is said to be \cbl{recursively enumerable} (r.e.) if there is some Turing machine $M$ that terminates successfully on input $x$ if and only if $x \in L$.
\item A set $L$ is said to be \cbl{co-recursively enumerable} (co-r.e.) if there is some Turing machine $M$ that terminates successfully on input $x$ if and only if $x \not \in L$.
\item A set is recursive iff it is r.e.~and co-r.e.
\end{itemize}
\end{frame}


\begin{frame}
\begin{itemize}
\item Turing showed that the set of \cbl{satisfiable}
formulas of first-order logic is not r.e.
\item Trakhtenbrot showed that the set of \cbl{finitely satisfiable}
formulas of first-order logic is not co-r.e.
\end{itemize}
\end{frame}


\begin{frame}
	\begin{itemize}	
		\item An algorithmic solution of the
		{\em Entscheidungsproblem} was already known for certain 
		restricted classes of first-order formulas, or \cbl{fragments} of first-order logic:\\
		\ \\
		\begin{itemize}
			\item The monadic fragment (L.~L\"{o}wenheim, 1915)
			\item Formulas $\exists \bar{x} \forall \bar{y}\, \psi$ (P.~Bernays and M.~Sch\"{o}nfinkel, 1928)
			\item Formulas $\exists \bar{x} \forall y \exists \bar{z}\, \psi$ (W.~Ackermann, 1928)
			\item Equality-free formulas $\exists \bar{x} \forall y_1 \forall y_2 \exists \bar{z}\, \psi$ 	
			(K.~G\"{o}del, K.~Sch\"{u}tte, 
			L.~Kalm\'{a}r~1933).
		\end{itemize}.
		\item Thus, Hilbert and Ackermann's \textit{Entscheidungsproblem} becomes a \cbl{classification problem}: which fragments of first-order logic are decidable for validity?
	\end{itemize}
\end{frame}



\begin{frame}
	\begin{itemize}
		\item Thus, if $\cL$ is any fragment of first-order logic,
		the \cbl{satisfiability problem} for $\cL$ is the following problem: 
		\problem{\Sat(\cL)}{a finite set $\Phi$ of $\cL$-formulas}{if $\Phi$ is satisfiable}{otherwise.}
		\item The \cbl{finite satisfiability problem} for $\cL$ is the following problem: 
		\problem{\FinSat(\cL)}{a finite set $\Phi$ of $\cL$-formulas}{if $\Phi$ is finitely satisfiable}{otherwise.}
		\item Most fragments of interest are closed under conjunction. In that case, we may take the input to $\Sat(\cL)$ or $\FinSat(\cL)$ to be a single formula $\phi$.
		\item We ask: for which fragments $\cL$ of first-order logic  are $\Sat(\cL)$ and/or $\FinSat(\cL)$ decidable.
	\end{itemize}
\end{frame}

\begin{frame}
\begin{itemize}
\item The principal fragments of first-order logic which are decidable
for satisfiability and finite satisfiability fall into four major classes:
\begin{itemize}
	\item two-variable fragments;
	\item the guarded fragments;
	\item quantifier-prefix fragments;
	\item variable-ordering fragments (the `fluted' fragment).
\end{itemize} 
\item 
The above fragments include all the most venerable logical system of all---Aristotle's syllogistic.
\item In this series of lectures, we shall be explaining the main ideas behind the decision procedures for representative fragments in these classes.
\end{itemize}
\end{frame}

\begin{frame}
\begin{itemize}
\item 
A fragment $\cL$ of first-order logic is said to have the \cbl{finite model property} if all satisfiable finite sets of formulas
are finitely satisfiable, or, equivalently,
if $\Sat(\cL) = \FinSat(\cL)$. 
\item In 1930, K.~G\"{o}del showed that the set of valid sentences of first-order logic can be \cbl{axiomatically characterized}
in terms of a finite list of axiom schemas and the inference rules of generalization and \textit{modus ponens}.
\item 
Hence, every fragment of first-order logic with the finite model property has a decidable satisfiability problem: given a finite set of formulas $\Phi$, we can just enumerate the proofs and the finite structures in parallel and wait until we obtain either a model of $\Phi$ or a proof of $\neg 
\bigwedge \Phi$.
%\item The converse is not true: there are fragments which lack the FMP, but whose satisfiability and finite satisfiability problems are decidable. 
\end{itemize}
\end{frame}

\begin{frame}
\begin{itemize}
\item If a problem is algorithmically
solvable, the obvious question to ask is: how quickly? 
\item We make use of the familiar hierarchy of complexity classes
\begin{multline*}
\NLogSpace \subseteq \PTime \subseteq \NPTime \subseteq \PSpace \subseteq\\ 
  \ExpTime \subseteq \NExpTime{} \subseteq \cdots \hspace {0.125cm}
\end{multline*}
\item In particular, if $\Sat(\cL)$ or $\FinSat(\cL)$ decidable, we
ask: in what complexity classes do these problems lie.
\item For any complexity class  $\cC$, a problem $P$ is \cbl{$\cC$-hard}
if, any other problem $Q$ in $\cC$ can be reduced to it (many-one log-space reduction). 
\item A problem $P$ is \cbl{$\cC$-complete} if it is in $\cC$ and $\cC$-hard.
\end{itemize}
\end{frame}

\begin{frame}
\begin{itemize}
\item We observe a trade-off between expressive power and complexity/decidability:
\begin{itemize}
	\item The satisfiability problem for
	   the \cbl{Aristotelian syllogistic} is \NLogSpace-complete.
	\item The satisfiability problems for
	the decidable \cbl{quantifier-prefix fragments} mentioned above are all 
	\NExpTime-complete.
	\item The satisfiablility problem for the \cbl{fluted fragment} is decidable, but not in any class in the above
	hierarchy!
\end{itemize}
\item Eventually, we reach undecidability:  $\FinSat(\FO)$ is r.e.-complete  and
 $\Sat(\FO)$ is co-r.e.-complete. 
\end{itemize}
\end{frame}
